This is a simple "hello world" program in Go.

To ease us into the Zig way of doing things, we will use a buffered writer instead of the usual default. 

        
package main

import (
	"bufio"
	"os"
)

func main() {
	writer := bufio.NewWriter(os.Stdout)

	writer.WriteString("Hello, ")
	writer.WriteString("World!\n")
	writer.Flush()
}

In Zig, we must supply a buffer to the writer. We will stack-allocate this buffer by declaring var stdout_buffer: [1_024]u8 = undefined; then we pass this uninitialized memory

There is a distinct seperation between the implementation of the writer stdout_writer: File.Writer and the generic interface that wraps the implementation writer: *Io.Writer. 

        
const std = @import("std");
const Io = std.Io;
const File = std.Io.File;

pub fn main(init: std.process.Init) !void {
    var stdout_buffer: [1_024]u8 = undefined;

    var stdout_writer: File.Writer =
        File.stdout().writer(init.io, &stdout_buffer);

    const writer: *Io.Writer = &stdout_writer.interface;

    _ = try writer.write("Hello, ");
    _ = try writer.write("World!\n");

    try writer.flush();
}
